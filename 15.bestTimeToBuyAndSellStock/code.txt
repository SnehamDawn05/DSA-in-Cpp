class Solution {
public:
    int maxProfit(vector<int>& prices) {
        int bp= prices[0], sp=prices[1], profit=0;
        for(int i=1;i<prices.size();i++){
            sp=prices[i];
            profit=max(sp-bp,profit);
            bp=min(bp,prices[i]);
        }
        if(profit>0) return profit;
        else return 0;
    }
};
