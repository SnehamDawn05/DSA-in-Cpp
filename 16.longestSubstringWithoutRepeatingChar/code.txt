class Solution {
public:
    int lengthOfLongestSubstring(string s) {
        int i=0,j=0, l=0;
        unordered_set<int> seen;
        while(i<=j && j<s.length()){
            if(seen.find(s[j])==seen.end()){
                seen.insert(s[j]);
                j++;
                l= max(l, (int)seen.size());
            }else{
                seen.erase(s[i]);
                i++;
            }
        }
        return l;
    }
};
